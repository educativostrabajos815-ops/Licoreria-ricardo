        function loadDataFromFirebase() {
            if(!window.db || !window.userId || !window.onSnapshotFb) return;
            const docRef = window.docFb(window.db, 'artifacts', window.appId, 'users', window.userId, 'pos_data_v1', 'store_data');
            
            window.onSnapshotFb(docRef, (docSnap) => {
                if (docSnap.exists()) {
                    const data = docSnap.data();
                    const cloudTime = data.lastUpdated || 0;
                    const localTime = window.lastUpdated || 0;

                    if (cloudTime > localTime) {
                        window.inventory = data.inventory || [];
                        window.currentIdCounter = data.currentIdCounter || 1;
                        window.appConfig = data.appConfig || window.appConfig;
                        window.lastUpdated = cloudTime;
                        
                        window.salesHistory = (data.salesHistory || []).map(s => ({...s, date: safeDateParse(s.date)}));
                        window.gastosHistory = (data.gastosHistory || []).map(g => ({...g, date: safeDateParse(g.date)}));

                        localStorage.setItem('pos_licoreria_data', JSON.stringify(data));

                        if (window.cart.length > 0) {
                            window.cart.forEach(c => {
                                const updatedProd = window.inventory.find(p => p.id == c.product.id);
                                if (updatedProd) c.product = updatedProd;
                            });
                            updateCartUI();
                        }
                        renderActiveViewOnly();

                } else if (window.lastUpdated > 0) { saveDataToCloudOnly(); }

        function addToCart(productId) {
            const p = window.inventory.find(x => x.id == productId); const item = window.cart.find(x => x.product?.id == productId); if (!p) return;
            if ((item ? item.quantity : 0) >= p.stock) { const card = document.getElementById(`prod-card-${productId}`); if(card) { card.classList.add('animate-[shake_0.4s_ease-in-out]'); setTimeout(() => card.classList.remove('animate-[shake_0.4s_ease-in-out]'), 400); } return; }
            if (item) item.quantity++; else window.cart.push({ product: p, quantity: 1 });
            updateCartUI(); renderPosProducts(document.getElementById('searchInput').value);
        }
        
        function changeQuantity(productId, delta) {
            const index = window.cart.findIndex(x => x.product?.id == productId);
            if (index > -1) { if (delta > 0 && window.cart[index].quantity >= window.cart[index].product.stock) return; window.cart[index].quantity += delta; if (window.cart[index].quantity <= 0) window.cart.splice(index, 1); updateCartUI(); renderPosProducts(document.getElementById('searchInput').value); }
        }
